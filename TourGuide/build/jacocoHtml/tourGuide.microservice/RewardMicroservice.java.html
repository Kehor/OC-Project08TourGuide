package tourGuide.microservice;

import gpsUtil.location.VisitedLocation;
import tourGuide.proxy.RewardProxy;
import tourGuide.user.User;
import tourGuide.user.UserReward;

import java.util.List;
import java.util.UUID;


public class RewardMicroservice {

    private RewardProxy rewardProxy;

    public List<UserReward> getuserReward(UUID userID, List<VisitedLocation> visitedLocations) {

        List<UserReward> userRewards = rewardProxy.getuserReward(userID.toString(), visitedLocations);
        return userRewards;
    }
}
